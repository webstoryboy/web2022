sort()

sort() 메서드는 배열 요소를 정렬하고, 정렬된 배열을 반환합니다.

특징 설명
유형(type) Array.prototype.sort()
버전(version) ES1
결과값(return) 배열(array)
사용성 ★★★☆☆

문법(Syntax)

"배열".sort() 

const arrNum2 = [100, 200, 300, 400, 500];
const arrNumSort2 = arrNum2.sort(); //100,200,300,400,500

const arrNum3 = [100, 200, 300, 400, 500];
const arrNumSort3 = arrNum3.sort(function(a,b){return b-a}); //500,400,300,200,100

const arrNum4 = [500, 400, 300, 200, 100];
const arrNumSort4 = arrNum4.sort(function(a,b){return a-b}); //100,200,300,400,500

const arrNum5 = ['c', 'd', 'e', 'a', 'b'];
const arrNumSort5 = arrNum5.sort((a,b) => a.localeCompare(b)); //a,b,c,d,e

const arrNum6 = ['c', 'd', 'e', 'a', 'b'];
const arrNumSort6 = arrNum6.sort((a,b) => b.localeCompare(a));  //e,d,c,b,a
